import matplotlib.pyplot as plt # Data points (x = cost, y = team size/power) tools = { "Buffer": (0.1, 0.2), "Zoho Social": (0.3, 0.4), "Later": (0.5, 0.6), "Hootsuite": (0.7, 0.8), "Sprout Social": (0.9, 0.9), } # Plot setup fig, ax = plt.subplots(figsize=(7,7)) # Scatter points for tool, (x, y) in tools.items(): ax.scatter(x, y, s=120, marker="o", edgecolor="black", alpha=0.8) ax.text(x+0.02, y, tool, fontsize=11, va="center") # Axis labels ax.set_xlabel("Low Cost ⟶ High Cost", fontsize=12, labelpad=15) ax.set_ylabel("Simple / Small Teams ⟶ Powerful / Large Teams", fontsize=12, labelpad=15) # Title ax.set_title("Social Media Automation Tool Landscape", fontsize=14, fontweight="bold", pad=20) # Add grid and quadrant feel ax.grid(True, linestyle="--", alpha=0.5) # Set limits for better spacing ax.set_xlim(0, 1) ax.set_ylim(0, 1) plt.show()
